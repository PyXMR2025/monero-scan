My signatures are now in binary format.

I made a small bash script I use on macOS to automate the process, I post it here in case someone's interested. The gitian-sigs repo local copy is located in ~/GitHub/gitian.sigs

```
#!/bin/bash
# init
function pause(){
   read -p "$*"
}

clear
echo "Script to sign .assert files in gitian-sigs"
echo "============================================================================"
cd ~/GitHub/gitian.sigs
echo "List of versions in ~/GitHub/gitian.sigs/ : "
ls -d1 v0* | cut -c 1-9 | sort -u
echo ""

# Assign most recent version to MONERO_VERSION
MONERO_VERSION=`ls -d1 v0* | cut -c 1-9 | sort -u | tail -1`
pause "Insert the version to sign or press Enter to accept the last ${MONERO_VERSION}: "

# If user input is not plain Enter, put user input in MONERO_VERSION
# WARNING!!! User input string is not tested!
if [ -n "$REPLY" ] ; then
MONERO_VERSION=${REPLY}
fi

GH_USER=stefanomarty

echo ""
echo "Signing based on the following parameters"
echo "GH_USER: ${GH_USER}"
echo "MONERO_VERSION: ${MONERO_VERSION}"
pause "Enter to continue, CTRL-C to abort"

# Here we could create a while cicle based on the various operating systems (android, linux, ecc.)
cd ~/GitHub/gitian.sigs/${MONERO_VERSION}-android/${GH_USER}
echo ""
echo "Signing assert in ~/GitHub/gitian.sigs/${MONERO_VERSION}-android/${GH_USER}"
gpg --detach-sig *.assert

cd ~/GitHub/gitian.sigs/${MONERO_VERSION}-freebsd/${GH_USER}
echo ""
echo "Signing assert in ~/GitHub/gitian.sigs/${MONERO_VERSION}-freebsd/${GH_USER}"
gpg --detach-sig *.assert

cd ~/GitHub/gitian.sigs/${MONERO_VERSION}-linux/${GH_USER}
echo ""
echo "Signing assert in ~/GitHub/gitian.sigs/${MONERO_VERSION}-linux/${GH_USER}"
gpg --detach-sig *.assert

cd ~/GitHub/gitian.sigs/${MONERO_VERSION}-osx/${GH_USER}
echo ""
echo "Signing assert in ~/GitHub/gitian.sigs/${MONERO_VERSION}-osx/${GH_USER}"
gpg --detach-sig *.assert

cd ~/GitHub/gitian.sigs/${MONERO_VERSION}-win/${GH_USER}
echo ""
echo "Signing assert in ~/GitHub/gitian.sigs/${MONERO_VERSION}-win/${GH_USER}"
gpg --detach-sig *.assert

echo ""
echo "SIGNING COMPLETED"
`
